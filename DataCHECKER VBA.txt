Option Compare Text

Dim wb2 As Workbook

Public Sub Allworkbooks()


Application.ScreenUpdating = False

Dim Sheet As Worksheet
Dim PasteStart As Range
Dim wnd As Window
Dim lt As Single, tp As Single, wd As Single, ht As Single

Set wb1 = ActiveWorkbook
Set PasteStart = [DataChecker!A1]
Set wnd = Application.ActiveWindow

   WindowState = xlMaximized
   ActiveWindow.WindowState = xlMaximized

    lt = ActiveWindow.Left
    wd = ActiveWindow.Width / 2
    ht = ActiveWindow.Height
    tp = ActiveWindow.Top
    
    ActiveWindow.WindowState = xlNormal
    ActiveWindow.Left = lt
    ActiveWindow.Top = tp
    ActiveWindow.Width = wd
    ActiveWindow.Height = ht

On Error Resume Next
Sheets("DataChecker").Unprotect
Sheets("DataChecker").Visible = True

Sheets("DataChecker").Select
   Cells.Select
   Selection.ClearFormats

Application.GoTo Sheets("DataChecker").Range("A1"), True
  
FileToOpen = Application.GetOpenFilename _
(Title:="Please choose your work plan", _
FileFilter:="All Files (*.*),*.*")

If FileToOpen = False Then
    MsgBox "No File Specified.", vbExclamation, "ERROR"
    Sheets("DataChecker").Visible = False
    Sheets("Error Report").Visible = False
    Exit Sub
     Else
      
    Set wb2 = Workbooks.Open(FileName:=FileToOpen)
    
    Application.GoTo Sheets("Performance Measures").Range("A1"), True
    Windows.CompareSideBySideWith wnd.Caption


    ActiveWindow.WindowState = xlNormal
    ActiveWindow.Width = wd
    ActiveWindow.Left = lt + wd
    ActiveWindow.Height = ht
    ActiveWindow.Top = tp
  
  
    Application.GoTo Sheets("Sheet1").Range("A1"), True
    
    wb2.Sheets("Sheet1").UsedRange.Copy
    PasteStart.Parent.Activate
    PasteStart.PasteSpecial xlPasteValuesAndNumberFormats
    PasteStart.PasteSpecial xlPasteFormats


    
    
    Call CheckValidations
    
      End If

End Sub

Public Sub OpenFile()

Set wb1 = ActiveWorkbook
Set PasteStart = [DataChecker!A1]
Set wnd = Application.ActiveWindow

Application.ScreenUpdating = False

Sheets("DataChecker").Unprotect

   Cells.Select
   Selection.ClearContents
    
 On Error GoTo Message
     
    wb2.Sheets("Sheet1").UsedRange.Copy

    PasteStart.PasteSpecial xlPasteFormats

Exit Sub

Message:
MsgBox "Your Work Plan is no longer open. Please use the Home tab to reopen."

 End Sub
